#Encapsulation and Private Attributes
class BankAccount:
    def __init__(self, owner, balance):
        self.owner = owner
        self.__balance = balance  # Private attribute

    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount
            return f"Deposited {amount}. New balance is {self.__balance}."
        return "Invalid deposit amount!"

    def withdraw(self, amount):
        if amount <= self.__balance:
            self.__balance -= amount
            return f"Withdrew {amount}. New balance is {self.__balance}."
        return "Insufficient funds!"

# Creating an instance
account = BankAccount("John", 1000)

# Accessing methods
print(account.deposit(500))  # Output: Deposited 500. New balance is 1500.
print(account.withdraw(200))  # Output: Withdrew 200. New balance is 1300.

Deposited 500. New balance is 1500.
Withdrew 200. New balance is 1300.

#Polymorphism and Method Overriding
class Bird:
    def sound(self):
        return "Some generic bird sound"

class Sparrow(Bird):
    def sound(self):
        return "Sparrow chirps"

class Peacock(Bird):
    def sound(self):
        return "Peacock screams"

def make_bird_sound(bird):
    print(bird.sound())

# Creating objects
sparrow = Sparrow()
peacock = Peacock()

# Demonstrating polymorphism
make_bird_sound(sparrow)  # Output: Sparrow chirps
make_bird_sound(peacock)  # Output: Peacock screams
